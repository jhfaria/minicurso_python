log_b10 = math.log10(x)		# log(x) na base 10
pot 	= math.pow(x, y)	# x^y

asen	= math.asin(x)		# arco seno (em radianos)

cos		= math.cos(x)		# cosseno (x em rad!)
sen		= math.sin(x)		# seno (x em rad!)

deg		= math.degrees(x)	# converte de radianos para graus
rad 	= math.radians(x)	# converte de graus para radianos
